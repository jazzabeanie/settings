# The following code replaces cd with cd then ls -lah. I found it here: https://askubuntu.com/questions/16106/how-can-i-create-an-alias-for-cd-and-ls 
function cd() {
    	new_directory="$*";
    	if [ $# -eq 0 ]; then 
		new_directory=${HOME};
	fi;
	builtin cd "${new_directory}" && ls -lah
}
